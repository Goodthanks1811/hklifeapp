// Function to check if at least one radio button is checked for each input name and compare checked count with group count
function checkAtLeastOneRadioButtonCheckedAndCompareCounts() {
  const radioButtons = form.querySelectorAll('input[type="radio"]:not([disabled])');
  const radioGroups = {};

  for (let i = 0; i < radioButtons.length; i++) {
    const radioButton = radioButtons[i];
    const name = radioButton.name;

    // Skip disabled radio buttons
    if (radioButton.disabled) {
      continue;
    }

    // If the name is not in the object, add it and initialize the count
    if (!radioGroups[name]) {
      radioGroups[name] = {
        total: 0,
        checked: 0
      };
    }

    // Increment the total count for the group
    radioGroups[name].total++;
  }

  for (const groupName in radioGroups) {
    const group = radioGroups[groupName];
    const radioButtonsWithName = form.querySelectorAll(`input[type="radio"][name="${groupName}"]:not([disabled])`);

    for (let i = 0; i < radioButtonsWithName.length; i++) {
      if (radioButtonsWithName[i].checked) {
        // Increment the checked count for the group
        group.checked++;
        break; // Found at least one checked radio button, break the loop
      }
    }
  }

  for (const groupName in radioGroups) {
    const group = radioGroups[groupName];

    // If no radio button is checked for a group, return false
    if (group.checked === 0) {
      return false;
    }
  }

  // Compare checked count with total group count
  const groupCount = Object.keys(radioGroups).length;
  let checkedCount = 0;

  for (const groupName in radioGroups) {
    const group = radioGroups[groupName];
    checkedCount += group.checked;
  }

// returns true or false if all radio buttons have been selected.
  return checkedCount === groupCount;
}

function hideAndEnableMenusOnMobile() {
  var mainMenu = document.getElementById("main-menu");

    mainMenu.classList.add("mobile-hidden");
    footerMenu.classList.add("footer-bar-desktop");
  } else {
    mainMenu.classList.remove("mobile-hidden");
    footerMenu.classList.remove("footer-bar-desktop");
  }
}
