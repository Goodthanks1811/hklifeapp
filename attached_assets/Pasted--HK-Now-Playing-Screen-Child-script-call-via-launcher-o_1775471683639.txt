<script>
// ── State ──────────────────────────────────────────
let playing = false;
let secs = ${t.elapsed};
let total = ${t.duration};
let hearted = ${t.liked};
let shuffle = true;
let repeat = false;

const queue = [
  {n:'2', t:'In Da Club Remix 2022',       a:'2Pac',        d:'4:12'},
  {n:'3', t:'Shape of You (Reggae Flip)',   a:'Ed Sheeran',  d:'3:55'},
  {n:'4', t:'Heartless Reggae Dub Cover',   a:'Kanye West',  d:'3:28'},
];

// ── Helpers ────────────────────────────────────────
function fmt(s) {
  return Math.floor(s/60) + ':' + String(s%60).padStart(2,'0');
}

function updateScrub() {
  document.getElementById('scrubFill').style.width = Math.round((secs/total)*100) + '%';
  document.getElementById('tElapsed').textContent = fmt(secs);
}

// ── Queue ──────────────────────────────────────────
document.getElementById('qlist').innerHTML = queue.map(t =>
  '<div class="qrow"><span class="qnum">'+t.n+'</span><div class="qinfo"><div class="qt">'+t.t+'</div><div class="qa">'+t.a+'</div></div><span class="qdur">'+t.d+'</span></div>'
).join('');

// ── Controls ───────────────────────────────────────
function togglePlay() {
  playing = !playing;
  document.getElementById('playIcon').innerHTML = playing
    ? '<rect x="5" y="4" width="4" height="16" rx="2" fill="#fff"/><rect x="15" y="4" width="4" height="16" rx="2" fill="#fff"/>'
    : '<polygon points="7,4 20,12 7,20" fill="#fff"/>';
}

function toggleHeart() {
  hearted = !hearted;
  const p = document.getElementById('hpath');
  p.setAttribute('fill', hearted ? '#ff1e1e' : 'none');
  p.setAttribute('stroke', hearted ? 'none' : '#555');
  p.setAttribute('stroke-width', hearted ? '0' : '1.5');
}

function toggleShuffle() {
  shuffle = !shuffle;
  document.getElementById('shuffleBtn').style.color = shuffle ? '#ff1e1e' : '#3a3a3a';
}

function toggleRepeat() {
  repeat = !repeat;
  document.getElementById('repeatBtn').style.color = repeat ? '#ff1e1e' : '#3a3a3a';
}

function prevTrack() {
  secs = 0; updateScrub();
}

function nextTrack() {
  secs = 0; updateScrub();
}

function scrub(e) {
  const r = document.getElementById('scrubTrack').getBoundingClientRect();
  secs = Math.round(((e.clientX - r.left) / r.width) * total);
  updateScrub();
}

function setVol(e) {
  const r = document.getElementById('volTrack').getBoundingClientRect();
  document.getElementById('volFill').style.width = Math.round(((e.clientX - r.left) / r.width) * 100) + '%';
}

function dismiss() {
  window.location.href = 'nav://back';
}

// ── Ticker ─────────────────────────────────────────
setInterval(() => {
  if (!playing) return;
  secs = (secs + 1) % total;
  updateScrub();
}, 1000);

// ── Viewport fix (canonical HK pattern) ───────────
function onVP() {
  const vp = window.visualViewport;
  if (!vp) return;
  document.body.style.top = vp.offsetTop + 'px';
  document.body.style.height = vp.height + 'px';
}
if (window.visualViewport) {
  window.visualViewport.addEventListener('resize', onVP);
  window.visualViewport.addEventListener('scroll', onVP);
  onVP();
}
<\/script>
</body>
</html>`;

async function run() {
  const wv = new WebView();
  await wv.loadHTML(buildHTML());
  await wv.present(true);
}

module.exports = { buildHTML, run };
await run();
