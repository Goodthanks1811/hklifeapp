var SWIPE_REVEAL=96;
function bindSwipeDelete(item){
  var track=item.querySelector('.auto-item-track');
  if(!track) return;
  var isOpen=false;
  var startX=0,startY=0,baseX=0;
  var dragging=false,decided=false,isHoriz=false;
  var lastX=0,lastT=0,vel=0;
  function getCardW(){ var inner=item.querySelector('.item-inner'); return inner?inner.offsetWidth:item.offsetWidth; }
  function setTrackX(x,dur){
    track.style.transition=dur!=null?('transform '+dur+'ms '+SWIPE_EASE):'none';
    track.style.transform='translateX('+x+'px)';
  }
  function snapOpen(v){
    isOpen=true; __openSwipeRow=item;
    var dur=v?Math.max(160,Math.min(320,260-v*0.5)):280;
    setTrackX(-SWIPE_REVEAL,dur);
  }
  function snapClose(v){
    isOpen=false;
    if(__openSwipeRow===item) __openSwipeRow=null;
    var dur=v?Math.max(160,Math.min(320,260-v*0.5)):280;
    setTrackX(0,dur);
  }
  function closeOther(){
    if(__openSwipeRow&&__openSwipeRow!==item){
      var ot=__openSwipeRow.querySelector('.auto-item-track');
      if(ot){ ot.style.transition='transform 300ms '+SWIPE_EASE; ot.style.transform='translateX(0)'; }
      __openSwipeRow=null;
    }
  }
  function onStart(e){
    if(window.__isAddModalOpen) return; if(window.__reorderMode) return;
    if(e.target.closest('.emoji')||e.target.closest('.check')||e.target.closest('.epic-tag')) return;
    if(e.target.closest('.del-btn')) return;
    closeOther();
    var pt=e.touches?e.touches[0]:e;
    startX=pt.clientX; startY=pt.clientY;
    lastX=startX; lastT=Date.now(); vel=0;
    baseX=isOpen?-SWIPE_REVEAL:0;
    dragging=true; decided=false; isHoriz=false;
    track.style.transition='none';
  }
  function onMove(e){
    if(!dragging) return;
    var pt=e.touches?e.touches[0]:e;
    var dx=pt.clientX-startX; var dy=pt.clientY-startY;
    if(!decided){
      if(Math.abs(dx)<5&&Math.abs(dy)<5) return;
      isHoriz=Math.abs(dx)>Math.abs(dy)*1.1;
      decided=true;
    }
    if(!isHoriz) return;
    if(Math.abs(dx)>8) __cancelLongPress();
    e.preventDefault();
    var now=Date.now(); var dt=now-lastT;
    if(dt>0) vel=(pt.clientX-lastX)/dt;
    lastX=pt.clientX; lastT=now;
    var raw=baseX+dx;
    var target;
    if(raw>0){ target=raw*0.1; }
    else if(raw<-SWIPE_REVEAL){ target=-SWIPE_REVEAL+(raw+SWIPE_REVEAL)*0.15; }
    else{ target=raw; }
    track.style.transition='none';
    track.style.transform='translateX('+target+'px)';
  }
  function onEnd(e){
    if(!dragging) return; dragging=false;
    if(!isHoriz) return;
    var pt=e.changedTouches?e.changedTouches[0]:e;
    var dx=pt.clientX-startX;
    var totalX=baseX+dx;
    var absVel=Math.abs(vel)*1000;
    if(isOpen){
      if(vel>0.25||dx>25){ snapClose(absVel); } else { snapOpen(absVel); }
    } else {
      if(vel<-0.25||-totalX>25){ snapOpen(absVel); } else { snapClose(absVel); }
    }
  }
  var card=item.querySelector('.item-inner');
  if(card){
    card.addEventListener('touchstart',onStart,{passive:true});
    card.addEventListener('touchmove',onMove,{passive:false});
    card.addEventListener('touchend',onEnd,{passive:true});
    card.addEventListener('touchcancel',onEnd,{passive:true});
  }
  var delBtn=item.querySelector('.del-btn');
  if(delBtn){
    delBtn.addEventListener('click',function(e){
      e.preventDefault(); e.stopPropagation();

      track.style.transform='translateX(-'+(cardW+SWIPE_REVEAL)+'px)';
      setTimeout(function(){ performDelete(item.dataset.id); },220);
    });
    delBtn.addEventListener('touchend',function(e){ e.stopPropagation(); },{passive:true});
  }
}
