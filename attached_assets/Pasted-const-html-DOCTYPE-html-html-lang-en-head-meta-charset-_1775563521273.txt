<script>
var EASE = 'cubic-bezier(0.22, 1, 0.36, 1)';
var openRow = null;

var ROWS = [
  { w: 'w1', t: 't1', c: 'c1', d: 'd1', reveal: 96 },
  { w: 'w2', t: 't2', c: 'c2', d: 'd2', reveal: 96 },
  { w: 'w3', t: 't3', c: 'c3', d: 'd3', reveal: 88 },
  { w: 'w4', t: 't4', c: 'c4', d: 'd4', reveal: 88 },
];

function collapseAndRemove(wrapper) {
  var h = wrapper.offsetHeight;
  wrapper.style.transition = 'none';
  wrapper.style.height = h + 'px';
  wrapper.style.marginBottom = '0px';
  wrapper.style.opacity = '1';
  wrapper.offsetHeight;
  wrapper.style.transition = 'height 0.42s ' + EASE + ', margin-bottom 0.42s ' + EASE + ', opacity 0.3s ease';
  wrapper.style.height = '0';
  wrapper.style.marginBottom = '0';
  wrapper.style.opacity = '0';
  var section = wrapper.parentElement;
  if (section) {
    section.style.transition = 'margin-bottom 0.42s ' + EASE;
    section.style.marginBottom = '0';
  }
  setTimeout(function() {
    if (section) section.remove();
  }, 480);
}

function setX(track, x, dur) {
  track.style.transition = dur != null ? ('transform ' + dur + 'ms ' + EASE) : 'none';
  track.style.transform = 'translateX(' + x + 'px)';
}

function initRow(cfg) {
  var wrapper = document.getElementById(cfg.w);
  var track   = document.getElementById(cfg.t);
  var card    = document.getElementById(cfg.c);
  var delBtn  = document.getElementById(cfg.d);
  var REVEAL  = cfg.reveal;

  // Set card width from probe
  var probe = document.createElement('div');
  probe.style.cssText = 'width:100%;height:0;';
  wrapper.appendChild(probe);
  requestAnimationFrame(function() {
    var cardW = probe.getBoundingClientRect().width;
    wrapper.removeChild(probe);
    card.style.width = cardW + 'px';

    var isOpen = false;
    var startX = 0, startY = 0, baseX = 0;
    var dragging = false, decided = false, isHoriz = false;
    var lastX = 0, lastT = 0, vel = 0;

    delBtn.addEventListener('click', function() {
      openRow = null;
      track.style.transition = 'transform 200ms cubic-bezier(0.4,0,1,1)';
      track.style.transform = 'translateX(-' + (cardW + REVEAL) + 'px)';
      setTimeout(function() { collapseAndRemove(wrapper); }, 230);
    });

    function closeOther() {
      if (openRow && openRow !== wrapper) {
        var ot = openRow.querySelector('.row-track');
        if (ot) setX(ot, 0, 300);
        openRow = null;
      }
    }

    function snapOpen(v) {
      isOpen = true; openRow = wrapper;
      var dur = v ? Math.max(160, Math.min(320, 260 - v * 0.5)) : 280;
      setX(track, -REVEAL, dur);
    }

    function snapClose(v) {
      isOpen = false;
      if (openRow === wrapper) openRow = null;
      var dur = v ? Math.max(160, Math.min(320, 260 - v * 0.5)) : 280;
      setX(track, 0, dur);
    }

    function onStart(e) {
      closeOther();
      var pt = e.touches ? e.touches[0] : e;
      startX = pt.clientX; startY = pt.clientY;
      lastX = startX; lastT = Date.now(); vel = 0;
      baseX = isOpen ? -REVEAL : 0;
      dragging = true; decided = false; isHoriz = false;
      track.style.transition = 'none';
    }

    function onMove(e) {
      if (!dragging) return;
      var pt = e.touches ? e.touches[0] : e;
      var dx = pt.clientX - startX;
      var dy = pt.clientY - startY;
      if (!decided) {
        if (Math.abs(dx) < 5 && Math.abs(dy) < 5) return;
        isHoriz = Math.abs(dx) > Math.abs(dy) * 1.1;
        decided = true;
      }
      if (!isHoriz) return;
      e.preventDefault();
      var now = Date.now(); var dt = now - lastT;
      if (dt > 0) vel = (pt.clientX - lastX) / dt;
      lastX = pt.clientX; lastT = now;
      var raw = baseX + dx;
      var target;
      if (raw > 0) { target = raw * 0.1; }
      else if (raw < -REVEAL) { target = -REVEAL + (raw + REVEAL) * 0.15; }
      else { target = raw; }
      track.style.transition = 'none';
      track.style.transform = 'translateX(' + target + 'px)';
    }

    function onEnd(e) {
      if (!dragging) return;
      dragging = false;
      if (!isHoriz) return;
      var pt = e.changedTouches ? e.changedTouches[0] : e;
      var dx = pt.clientX - startX;
      var totalX = baseX + dx;
      var absVel = Math.abs(vel) * 1000;
      if (isOpen) {
        if (vel > 0.25 || dx > 25) { snapClose(absVel); } else { snapOpen(absVel); }
      } else {
        if (vel < -0.25 || -totalX > 25) { snapOpen(absVel); } else { snapClose(absVel); }
      }
    }

    card.addEventListener('touchstart', onStart, { passive: true });
    card.addEventListener('touchmove', onMove, { passive: false });
    card.addEventListener('touchend', onEnd);
    card.addEventListener('touchcancel', onEnd);
  });
}

ROWS.forEach(function(cfg) { initRow(cfg); });
<\/script>
</body>
</html>`;
